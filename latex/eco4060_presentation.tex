\section{Introduction}



\begin{frame}{Assessing macro-financial tail risks still challenging}

Standard macro-fin toolkit: GDP-at-risk \parencite{adrianVulnerableGrowth2019}; unemployment-at-risk \parencite{kileyUnemploymentRisk2022}; inflation-at-risk \parencite{lopez-salidoInflationRisk2024}
    \vspace{.4cm}

    \begin{enumerate}
        \item \textbf{How to fully capture non-linearity and possible skewness?}
        \begin{itemize}
            \item \textcite{adrianVulnerableGrowth2019}, \textcite{kileyUnemploymentRisk2022}, and \textcite{lopez-salidoInflationRisk2024} use quantile regression (conditional linear)
        \end{itemize}
        \vspace{.4cm}
        \item \textbf{How to allow for big data?}
        \begin{itemize}
            \item \textcite{adrianVulnerableGrowth2019} assume only an index of financial conditions matter
            \item \textcite{adrianMachinelearningGrowthRisk2025} and \textcite{chiMacroeconomicForecastingMachine2025} use Quantile ML
        \end{itemize}            
        \vspace{.4cm}
        \item \textbf{How to ensure coherent joint movement of tails?}
        \begin{itemize}
            \item Quantile VAR \parencite{chavleishviliForecastingStressTesting2024} %ensure same process drives all quantiles
            \item Modelling quantile spacing \parencite{schmidtQuantileSpacingsSimple2016} prevents quantile crossing
        \end{itemize}
        \vspace{.4cm}
    \end{enumerate}
    Challenges can be solved with our new ML method
\end{frame}

\begin{frame}{How do we assess macro-financial tail risks in this paper?}
    $\Rightarrow$ \textbf{Deep learning}: allows for high degree of time-varying complexity and high-dimensional data \\
    \vspace{.2cm}
    $\Rightarrow$ \textbf{Multi-task}: use info learned by each task to learn other tasks \parencite{caruanaMultitaskLearning1997} \\% reduces overfitting (Ruder 2017)\\
    \vspace{.2cm}
    $\Rightarrow$ \textbf{No-crossing architecture}: implement quantile spacings \parencite{schmidtQuantileSpacingsSimple2016}\\
    \vspace{.7cm}
\end{frame}



\begin{frame}{Contributions}

        \item We consider 5 quantiles and explore possible (shared) drivers
        \item We explore shorter and longer horizon (1 and 4 quarters ahead)
	\end{itemize}

\section{Data and Methodology}


\begin{frame}{Data and Methodology}
    \begin{itemize}
        \item 116 stationary predictors from FRED-MD over Jan 1961--Dec 2024.
        \item Response variables: 
            \begin{itemize}
                \item Inflation: $\pi_{t+h} := \ln(CPI_{t+h}) - \ln(CPI_{t})$.
                \item Industrial Production growth: $y_{t+h} := \ln(IP_{t+h}) - \ln(IP_{t})$.
                \item Pct change in unemployment rate: $u_{t+h} := \ln(UR_{t+h}) - \ln(UR_{t})$.
            \end{itemize} 
        \item Annual expanding window training procedure beginning with Jan 1961--Dec 1997 to forecast Jan 1998--Dec 1998.
    \end{itemize}
\end{frame}

\begin{frame}{Data and Methodology}
    \begin{itemize}
        \item Quantile forecasts are evaluated with the out-of-sample $R^1$ metric: 
        $$
        R^1_{oos} = 1- \frac{\sum_{t=1}^{T_{oos}} \rho_\tau (y_{t+h} - \hat y_{\tau,t+h}) }{\sum_{t=1}^{T_{oos}} \rho_\tau (y_{t+h} - \bar y_{\tau}) }
        $$
        where $ \bar y_{\tau}$ is the expanding historical quantile
        \vspace{.2cm}
        \item Mean forecasts are evaluated with the out-of-sample $R^2$ metric:
        $$
        R^2_{oos} = 1- \frac{\sum_{t=1}^{T_{oos}}  (y_{t+h} - \bar Q_{t+h})^2 }{\sum_{t=1}^{T_{oos}} (y_{t+h} - \bar y)^2 }
        $$
        where $\bar y$ is the expanding historical mean \\and $\bar Q$ is the robust estimate of the mean \parencite{liuStockDistributionRisk2023}:
        $$
        \bar Q = 0.15 Q_{0.05} + 0.225Q_{0.25} + 0.25Q_{0.5} + 0.75Q_{0.75} + 0.15Q_{0.95}
        $$
    \end{itemize}    

\end{frame}

\section{Appendix}
\begin{frame}[allowframebreaks]{Theorem 1 (Quantile Mean Formula)}
\textbf{Theorem 1.} Let $F_t(x) = P(r_{i,t+1} \leq x \mid \mathcal{F}_t)$ be the cumulative distribution function of stock excess return $r_{i,t+1}$ conditional on information set $\mathcal{F}_t$.

For $0 < \tau_1 < \cdots < \tau_K < 1$, let
\[
\begin{aligned}
\bar{Q}_K &= (\tau_1 + \frac{\tau_2 - \tau_1}{2}) Q_{\tau_1} + (\frac{\tau_2 - \tau_1}{2} + \frac{\tau_3 - \tau_2}{2}) Q_{\tau_2} + \cdots \\
&\quad + (\frac{\tau_{K-2} - \tau_{K-1}}{2} + \frac{\tau_{K-1} - \tau_K}{2}) Q_{\tau_{K-1}} + (\frac{\tau_{K} - \tau_{K-1}}{2} + (1-\tau_K)) Q_{\tau_K} \\
&= \frac{\tau_2 + \tau 1}{2}Q_{\tau_1} + \frac{\tau_3 - \tau_1}{2} Q_{\tau_2} + ... + \frac{\tau_K - \tau_{K-2}}{2} Q_{\tau_{K-1}} + (1 - \frac{\tau_{K} - \tau_{K-1}}{2}) Q_{\tau_K} \\
\end{aligned}
\]
where $Q_{\tau_k} = Q_{\tau_k}(r_{i,t+1} \mid \mathcal{F}_t)$ are quantiles of $r_{i,t+1}$, $k = 1, \ldots, K$.

Assume that $F_t(x)$ is continuous and strictly monotonically increasing, $E_t(r_{i,t+1})$ exists, and
\[
E_t(r_{i,t+1}) = \int_{\mathbb{R}} x\, d\mu_{F_t},
\]
where $\int_{\mathbb{R}} x\, d\mu_{F_t}$ is a Lebesgue–Stieltjes integral, $\mu_{F_t}$ is the Lebesgue–Stieltjes measure associated with $F_t$, and $\mathbb{R}$ is the set of real numbers. Then
\[
\lim_{\Delta_1 \to 0,\, \Delta_2 \to 0} \bar{Q}_K = E_t(r_{i,t+1}),
\]
where $\Delta_1 = \max\{\tau_2 + \tau_1, 1 - (\tau_K + \tau_{K-1})\}$ and $\Delta_2 = \max_{2 \leq k \leq K-1} \{ Q_{(\tau_k + \tau_{k+1})/2} - Q_{(\tau_{k-1} + \tau_k)/2} \}$.
\end{frame}

\begin{frame}{Proof of Theorem 1 (Sketch)}
\textbf{Proof.} Without loss of generality, fix firm $i$ and time $t$. Since $\int_{\mathbb{R}} x\, d\mu_{F_t}$ exists, then the Riemann–Stieltjes integral $\int_{\mathbb{R}} x\, dF_t(x)$ exists and $E_t(r_{i,t+1}) = \int_{\mathbb{R}} x\, dF_t(x)$.

Let $[a, b]$ be an interval containing the support of $r_{i,t+1}$, and partition $[a, b]$ into subintervals. Define the Riemann–Stieltjes sum
\[
RS_t(K) = \sum_{k=1}^K c_k (F_t(x_k) - F_t(x_{k-1})),
\]
where $c_k \in (x_{k-1}, x_k]$. As the partition gets finer, $RS_t(K) \to \int_a^b x\, dF_t(x)$.

By choosing quantile points and weights as in the theorem, the sum $\bar{Q}_K$ approximates $E_t(r_{i,t+1})$ arbitrarily well as the mesh of the partition and the quantile spacing go to zero. Thus, $\bar{Q}_K \to E_t(r_{i,t+1})$ as $K \to \infty$ and the quantile grid becomes dense.

\hfill$\square$
\end{frame}
